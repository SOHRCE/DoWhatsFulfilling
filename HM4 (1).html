def to_one_hot(y, num_class=10):
    #from keras.utils.np_utils import to_categorical   
    #categorical_labels = to_categorical(y, num_class)
    results = np.zeros((len(y), num_class))
    for i, label in enumerate(y):
      results[i, label] = 1.
    return results
